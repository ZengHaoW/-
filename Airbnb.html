def fun(x):
    s1 = place[x.neighbourhood_group]
    s2 = math.log(x.price, 10000)
    s3 = Type[x.room_type]
    if x.availability_365 > 365:
        s4 = 1
    else:
        s4 = x.availability_365 / 365
    if x.minimum_nights > 365:
        s5 = 0
    else:
        s5 = (365 - x.minimum_nights) / 365
    if x.number_of_reviews == 0:
        s6 = 0
    else:
        s6 = math.log(x.number_of_reviews, 629)
    return s2 * 0.2+ s1 / 3 * 0.2 + s3 / 3 * 0.2 + s4*0.1 + s5*0.1 + s6 * 0.2
rr = df.apply(fun,axis=1)
df.s = rr
